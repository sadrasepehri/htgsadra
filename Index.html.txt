// Simple SPA router (hash-based)
const routes = [];
document.querySelectorAll('[data-route]').forEach(s => routes.push(s));

function show(route) {
  routes.forEach(sec => sec.classList.remove('active'));
  const el = document.querySelector(route);
  (el ? el : document.querySelector('#home')).classList.add('active');
  // Close mobile menu on navigation
  const m = document.getElementById('menu');
  const h = document.getElementById('hamburger');
  m?.classList.remove('open');
  h?.setAttribute('aria-expanded','false');
  // Set document title per page for basic SEO
  const titleMap = {
    '#home': 'HTG Inc. — Nationwide Full-Lifecycle IT & Cybersecurity',
    '#services-lifecycle': 'IT Lifecycle & Logistics — HTG Inc.',
    '#services-managed': 'Managed IT & Cloud — HTG Inc.',
    '#services-security': 'Cybersecurity (MSSP) — HTG Inc.',
    '#estore': 'HTG e-Store — HTG Inc.',
    '#about': 'About HTG Inc.',
    '#industries': 'Industries — HTG Inc.',
    '#cases': 'Case Studies — HTG Inc.',
    '#blog': 'Blog — HTG Inc.',
    '#team': 'Meet the Team — HTG Inc.',
    '#support': 'Support Team — HTG Inc.',
    '#contact': 'Contact — HTG Inc.',
    '#book': 'Book a Consult — HTG Inc.'
  };
  document.title = titleMap[route] || 'HTG Inc.';
}

// Wire nav clicks
document.addEventListener('click', (e) => {
  const link = e.target.closest('[data-nav]');
  if (!link) return;
  e.preventDefault();
  const target = link.getAttribute('data-nav') || '#home';
  location.hash = target;
});

// Hash change
window.addEventListener('hashchange', () => show(location.hash || '#home'));

// Ready
document.addEventListener('DOMContentLoaded', () => {
  // Mobile menu
  const hamburger = document.getElementById('hamburger');
  const menu = document.getElementById('menu');
  hamburger?.addEventListener('click', () => {
    menu.classList.toggle('open');
    const expanded = hamburger.getAttribute('aria-expanded') === 'true';
    hamburger.setAttribute('aria-expanded', String(!expanded));
  });

  // Year
  const yearEl = document.getElementById('year');
  if (yearEl) yearEl.textContent = new Date().getFullYear();

  // Initial route
  show(location.hash || '#home');
});

// Contact form (demo)
document.addEventListener('submit', (e) => {
  const form = e.target.closest('#contactForm');
  if (!form) return;
  e.preventDefault();
  const email = document.getElementById('email').value.trim();
  const name = document.getElementById('name').value.trim();
  const help = document.getElementById('formHelp');
  if (!name || !email || !email.includes('@')) {
    help.textContent = 'Please enter your name and a valid email.';
    help.style.color = 'var(--danger)';
    return;
  }
  help.textContent = 'Thanks! We’ll reach out shortly.';
  help.style.color = 'var(--accent)';
  form.reset();
});
